\begin{description}
\item[Problem] Show that $\text{lcm}(a, b) = ab$ if and only if $\gcd(a,b) =
1$.

\item[Proof] Follows directly from the fact that $\text{lcm}(a, b) =
\dfrac{|ab|} {\gcd(a, b)}$:
\begin{align*}
|ab| = \text{lcm}(a, b) &\iff \\
|ab| = \frac{|ab|} {\gcd(a, b)} &\iff \\
\gcd(a, b) |ab| = |ab| &\iff \\
\gcd(a, b) = 1
\end{align*}

\end{description}
