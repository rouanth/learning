\begin{description}
\item[Problem]
\begin{itemize}
        \item Define a function $f : \mathbb{N} \rightarrow \mathbb{N}$ that
              is one-to-one but not onto.
        \item Define a function $f : \mathbb{N} \rightarrow \mathbb{N}$ that
              is onto but not one-to-one.
\end{itemize}
\item[Solution]
\begin{itemize}
        \item $f(n) = n + 1$
        \item $f(n) = \left\lfloor \frac n 2 \right\rfloor$
\end{itemize}
\end{description}
