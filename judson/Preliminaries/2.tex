\begin{description}
\item[Problem] If $A = \left\{a, b, c\right\}$, $B = \left\{1, 2, 3\right\}$,
$C = \left\{x\right\}$, and $D = \emptyset$, list all elements in each of the
following sets.

\begin{multicols}{2}
\begin{enumerate}
\item $A \times B$
\item $B \times A$
\item $A \times B \times C$
\item $A \times D$
\end{enumerate}
\end{multicols}

\item[Solution]

\begin{enumerate}
\item $A \times B = \left\{
(a, 1),
(a, 2),
(a, 3),
(b, 1),
(b, 2),
(b, 3),
(c, 1),
(c, 2),
(c, 3)
\right\}$
\item $B \times A = \left\{
(1, a),
(1, b),
(1, c),
(2, a),
(2, b),
(2, c),
(3, a),
(3, b),
(3, c)
\right\}$
\item $
A \times B \times C = \left\{
\begin{aligned}
(a, 1, x),&
(a, 2, x),
(a, 3, x),\\
(b, 1, x),&
(b, 2, x),
(b, 3, x),\\
(c, 1, x),&
(c, 2, x),
(c, 3, x)
\end{aligned}
\right\}$
\item $A \times D = \emptyset$
\end{enumerate}

\end{description}
